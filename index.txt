<!DOCTYPE html>
<html>
<head>
    <title>Live Image Stream</title>
</head>
<body>
    <img id="live-image" alt="Live Stream"/>
    <script>
        const socket = new WebSocket("ws://localhost:8080/image-stream");

        socket.onmessage = function(event) {
            console.log("Received message:", event.data); // Log received data
            const imageElement = document.getElementById("live-image");
            imageElement.src = 'data:image/png;base64,' + event.data; // Ensure correct MIME type
        };

        socket.onopen = function() {
            console.log("Connected to the server");
        };

        socket.onclose = function() {
            console.log("Disconnected from the server");
        };

        socket.onerror = function(error) {
            console.error("WebSocket error:", error);
        };
    </script>
</body>
</html>
