package com.example.healthcheck.service;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;
import java.util.List;

@Service
public class HealthCheckService {

    @Value("${services.urls}")
    private List<String> serviceUrls;

    private final RestTemplate restTemplate;

    public HealthCheckService(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }

    public long countServicesUp() {
        return serviceUrls.stream()
                .filter(this::isServiceUp)
                .count();
    }

    private boolean isServiceUp(String url) {
        try {
            String healthStatus = restTemplate.getForObject(url, String.class);
            return healthStatus != null && healthStatus.contains("\"status\":\"UP\"");
        } catch (Exception e) {
            return false;
        }
    }
}
