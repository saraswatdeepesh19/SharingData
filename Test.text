# Root logger configuration
status = warn
name = PropertiesConfig
property.logDir = logs
property.logFileName = ${logDir}/application.log

# Console appender
appender.console.type = Console
appender.console.name = Console
appender.console.layout.type = PatternLayout
appender.console.layout.pattern = %d{HH:mm:ss.SSS} [%t] %-5level %logger{36} - %msg%n

# File appender (writing logs to the 'logs/application.log' file)
appender.file.type = File
appender.file.name = LogFile
appender.file.fileName = ${logFileName}
appender.file.layout.type = PatternLayout
appender.file.layout.pattern = %d %p %c [%t] %m%n

# Root logger configuration (INFO level)
rootLogger.level = info
rootLogger.appenderRefs = console, file
rootLogger.appenderRef.console.ref = Console
rootLogger.appenderRef.file.ref = LogFile

# Set logging level for specific packages
logger.org.springframework.boot.autoconfigure.level = info

# Suppress debug-level logs from Spring Boot's autoconfiguration
logger.org.springframework.level = info
